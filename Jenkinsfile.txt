pipeline {
    agent any

    tools {
        nodejs "NodeJS 20" // match with Global Tools name
    }

    stages {
        stage('Clone') {
            steps {
                git 'https://github.com/Venkatadhi626/my-react-app.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }

        stage('Build React App') {
            steps {
                sh 'npm run build'
            }
        }

        stage('Deploy (Serve Locally)') {
            steps {
                sh 'npx serve -s build -l 5000'
            }
        }
    }
}
